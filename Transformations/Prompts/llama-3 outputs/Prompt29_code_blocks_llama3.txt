void spmv_parallel(int n, int nnz, float *val, int *row, int *col, float *x, float *y) {
  int i, j;
  #pragma hip parallel for default(none) shared(n, nnz, val, row, col, x, y) private(i, j)
  for (i = 0; i < n; i++) {
    y[i] = 0.0f;
    for (j = row[i]; j < row[i+1]; j++) {
      y[i] += val[j] * x[col[j]];
    }

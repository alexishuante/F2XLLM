void gemv(int n, float *A, float *x, float *y) {
    #pragma omp parallel for
    for (int i = 0; i < n; i++) {
        float sum = 0.0;
        for (int j = 0; j < n; j++) {
            sum += A[i*n + j] * x[j];
        }
        y[i] = sum;
    }
}

}

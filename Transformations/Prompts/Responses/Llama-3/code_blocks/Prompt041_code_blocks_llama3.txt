--------------- Code Block 1 ---------------
void saxpy_parallel(const int n, const float a, const float* x, float* y) {
  Kokkos::parallel_for(Kokkos::RangePolicy<>(0, n), 
    [=](const int i) {
      y[i] = a * x[i] + y[i];
    });
}

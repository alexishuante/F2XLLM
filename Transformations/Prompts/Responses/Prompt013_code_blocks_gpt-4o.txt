void gemv_parallel(int n, float A[n][n], float x[n], float y[n]) {
    int i, j;
    float sum;

    #pragma acc parallel loop private(sum) present(A, x, y)
    for (i = 0; i < n; i++) {
        sum = 0.0;
        for (j = 0; j < n; j++) {
            sum += A[i][j] * x[j];
        }
        y[i] = sum;
    }
}

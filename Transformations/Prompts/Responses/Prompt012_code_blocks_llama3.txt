void gemv_parallel(int n, double *A, double *x, double *y) {
    int i, j;
    double sum;

    #pragma acc data copy(A(:,:), x(:), y(:))
    #pragma acc parallel loop private(sum)
    for (i = 0; i < n; i++) {
        sum = 0.0;
        #pragma acc loop reduction(+:sum)
        for (j = 0; j < n; j++) {
            sum += A[i * n + j] * x[j];
        }
        y[i] = sum;
    }
}

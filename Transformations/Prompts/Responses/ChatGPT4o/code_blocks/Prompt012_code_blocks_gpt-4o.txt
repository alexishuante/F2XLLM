--------------- Code Block 1 ---------------
void saxpy(int n, float a, float *x, float *y) {
    // Parallelize the loop using OpenACC
    #pragma acc parallel loop
    for (int i = 0; i < n; i++) {
        y[i] = a * x[i] + y[i];
    }
}

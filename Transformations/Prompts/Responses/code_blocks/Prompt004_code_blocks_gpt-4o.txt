void gemv_parallel(int n, float **A, float *x, float *y) {
    int i, j;

    // Parallelize the outer loop with OpenMP
    #pragma omp parallel for private(i, j) shared(A, x, y)
    for (i = 0; i < n; i++) {
        float sum = 0.0;
        for (j = 0; j < n; j++) {
            sum += A[i][j] * x[j];
        }
        y[i] = sum;
    }
}

void gemv_parallel(int n, float *A, float *x, float *y) {
    int i, j;
    
    #pragma acc parallel loop private(j)
    for (i = 0; i < n; i++) {
        float sum = 0.0;
        #pragma acc loop reduction(+:sum)
        for (j = 0; j < n; j++) {
            sum += A[i * n + j] * x[j];
        }
        y[i] = sum;
    }
}

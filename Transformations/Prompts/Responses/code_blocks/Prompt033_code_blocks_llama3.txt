  cudaMalloc((void **)&d_A, n * n * sizeof(float));
  cudaMalloc((void **)&d_x, n * sizeof(float));
  cudaMalloc((void **)&d_y, n * sizeof(float));
  cudaMemcpy(d_A, A, n * n * sizeof(float), cudaMemcpyHostToDevice);

  cudaFree(d_y);
}

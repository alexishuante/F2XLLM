void gemv_serial(int n, float *A, float *x, float *y) {
  int i, j;
  float sum;
  for (i = 0; i < n; i++) {
    sum = 0.0;
    for (j = 0; j < n; j++) {
      sum += A[i * n + j] * x[j];
    }
    y[i] = sum;
  }
}
